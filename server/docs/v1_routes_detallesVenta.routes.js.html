v1/routes/detallesVenta.routes.js

import { Router } from "express";
import { createOneDetallesVenta, showAllDetallesVenta, showDetallesVentaById, updateOneDetallesVenta, deleteOneDetallesVenta } from "../../controllers/detallesVenta.controller.js";
import validateSchema from "../../middlewares/validateSchema.middleware.js";
import { saleDetailsSchema, updateSaleDetailsSchema } from "../../schemas/detallesVenta.schema.js";
import verifyToken from "../../middlewares/verifyToken.middlaware.js";
import rolesValidator from "../../middlewares/rolesValidator.middleware.js";

const router = Router();

/**
 * Ruta para obtener los detalles de una venta por ID.
 * 
 * @name GET /detalles-venta/:detallesVenta_Id
 * @function
 * @memberof module:routes/detallesVenta
 * @param {string} detallesVenta_Id - El ID de los detalles de venta a obtener.
 * @param {function} verifyToken - Middleware para verificar el token JWT.
 * @param {function} rolesValidator - Middleware para validar los roles de usuario.
 * @param {function} showDetallesVentaById - Controlador para obtener los detalles de la venta por ID.
 * @returns {Object} - Detalles de la venta con el ID especificado.
 */
router.get("/detalles-venta/:detallesVenta_Id", verifyToken, rolesValidator(["admin", "director"]), showDetallesVentaById);

/**
 * Ruta para obtener todos los detalles de ventas.
 * 
 * @name GET /detalles-venta
 * @function
 * @memberof module:routes/detallesVenta
 * @param {function} verifyToken - Middleware para verificar el token JWT.
 * @param {function} rolesValidator - Middleware para validar los roles de usuario.
 * @param {function} showAllDetallesVenta - Controlador para obtener todos los detalles de ventas.
 * @returns {Array<Object>} - Lista de todos los detalles de ventas.
 */
router.get("/detalles-venta", verifyToken, rolesValidator(["admin", "director"]), showAllDetallesVenta);

/**
 * Ruta para crear nuevos detalles de venta.
 * 
 * @name POST /detalles-venta
 * @function
 * @memberof module:routes/detallesVenta
 * @param {Object} data - Los datos de los nuevos detalles de venta.
 * @param {function} verifyToken - Middleware para verificar el token JWT.
 * @param {function} rolesValidator - Middleware para validar los roles de usuario.
 * @param {function} validateSchema - Middleware para validar el esquema de los detalles de venta.
 * @param {function} createOneDetallesVenta - Controlador para crear los detalles de venta.
 * @returns {Object} - Los detalles de venta creados.
 */
router.post("/detalles-venta", verifyToken, rolesValidator(["admin", "director"]), validateSchema(saleDetailsSchema), createOneDetallesVenta);

/**
 * Ruta para actualizar los detalles de una venta por ID.
 * 
 * @name PUT /detalles-venta/:detallesVenta_Id
 * @function
 * @memberof module:routes/detallesVenta
 * @param {string} detallesVenta_Id - El ID de los detalles de venta a actualizar.
 * @param {Object} data - Los datos actualizados de los detalles de venta.
 * @param {function} verifyToken - Middleware para verificar el token JWT.
 * @param {function} rolesValidator - Middleware para validar los roles de usuario.
 * @param {function} validateSchema - Middleware para validar el esquema de los detalles de venta actualizados.
 * @param {function} updateOneDetallesVenta - Controlador para actualizar los detalles de venta.
 * @returns {Object} - Mensaje de confirmación de la actualización.
 */
router.put("/detalles-venta/:detallesVenta_Id", verifyToken, rolesValidator(["admin", "director"]), validateSchema(updateSaleDetailsSchema), updateOneDetallesVenta);

/**
 * Ruta para eliminar detalles de venta.
 * 
 * @name DELETE /detalles-venta
 * @function
 * @memberof module:routes/detallesVenta
 * @param {Array<number>} ids - Los IDs de los detalles de venta a eliminar.
 * @param {function} verifyToken - Middleware para verificar el token JWT.
 * @param {function} rolesValidator - Middleware para validar los roles de usuario.
 * @param {function} deleteOneDetallesVenta - Controlador para eliminar los detalles de venta.
 * @returns {Object} - Mensaje de confirmación de la eliminación.
 */
router.delete("/detalles-venta", verifyToken, rolesValidator(["admin", "director"]), deleteOneDetallesVenta);

export default router;
